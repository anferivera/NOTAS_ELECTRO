{\color{red}LEY DE AMPERE}
\vspace{0.6 cm}

\begin{enumerate} 

\item Usando la ley de Ampere, calcule el campo magnético en el interior de solenoide muy largo de longitud $l$, de $N$ vueltas y de corriente $I$.
\begin{displaymath}
B=\mu_0 \dfrac{N}{l}I = \mu_0 n I \hspace*{0.5 cm};\hspace*{0.5 cm} \dfrac{N}{l}=n
\end{displaymath}

\item Usando la ley de Ampere, calcule el campo magnético dentro de un toroide de radio $r$.
\begin{displaymath}
B=\mu_0 \dfrac{NI}{2\pi r}
\end{displaymath}

\item A lo largo de un cilindro de radio $R$ fluye una corriente $I$ distribuida de modo uniforme. Calcule el campo magnético $\vec{B}$ en el interior y en el exterior del cable.
\begin{displaymath}
B=\dfrac{\mu_0}{2\pi}\dfrac{r}{R^2} \hspace*{1 cm} 0\ll r \ll R
\end{displaymath}
\begin{displaymath}
B=\dfrac{\mu_0}{2\pi} \dfrac{1}{r} \hspace*{1 cm} r\gg R
\end{displaymath}

\item Calcule el campo magnético de un cilindro de radio $R$ por el que fluye una corriente cuya densidad es $J=br$, donde $b$ es una constante y $r$ es la distancia radial al centro del cilindro.
\begin{displaymath}
B=\mu_0 b \dfrac{r^2}{3} \hspace*{1 cm} 0\ll r\ll R
\end{displaymath}
\begin{displaymath}
B=\mu_0 b \dfrac{R^3}{3}\dfrac{1}{r} \hspace*{1 cm} r\gg R
\end{displaymath}

\item Se tienen un cilindro de radio $R_{1}$ concéntrico con un cascarón cilíndrico de radio interno $R_{2}$ y radio externo $R_{3}$ tal como se muestra en la figura. El cilindro interior tiene una densidad de corriente $\vec{J}_{1}=b r \hat{k}$, donde $b$ es una constante y $r$ es la distancia al centro. El cascarón cilíndrico tiene una densidad de corriente constante $\vec{J}_{2}=-J_{2}\hat{k}$.\\

\begin{minipage}{7 cm}{
Hallar el campo magnético para:
\begin{enumerate}
\item $r \le R_{1}$
\item $R_{1} \le r \le R_{2}$
\item $R_{2} \le r \le R_{3}$
\item $r \geq R_{3}$
\end{enumerate}
}
\end{minipage}

\end{enumerate}
