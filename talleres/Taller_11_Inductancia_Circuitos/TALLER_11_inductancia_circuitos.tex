{\color{red}INDUCTANCIA}

\begin{enumerate} 
%ejercicio
\item  Determinar la inductancia de un solenoide de N vueltas, de longitud $l$ (grande) y área transversal $A$.
\begin{align}
L=\mu_0\dfrac{N^2}{l}A = \mu_0 n^2 l A = \mu_0 n^2 \text{Vol.} \nonumber
\end{align}

%ejercicio
\item  Determinar la inductancia de un toroide de N vueltas, de radio promedio $r$ y área transversal $A$.
\begin{align}
L=\mu_0\dfrac{N^2}{2\pi r}A  \nonumber
\end{align}

%ejercicio
\item Calcular la inductancia por unidad de longitud ($l$) de un cable coaxial muy largo. Suponga que son dos cubiertas cilíndricas de radios $a$ y $b>a$ con corrientes opuestas.
\begin{align*}
\dfrac{L}{l}=\mu_0\dfrac{1}{2\pi}\ln\left(\dfrac{b}{a}\right)
\end{align*}

{\color{red}CIRCUITOS LC}

%ejercicio
\item Considere un capacitor previamente cargado con una carga $\pm Q$ que se conecta en serie a un inductor. 
\begin{enumerate}
\item Determinar la carga en uno de los platos del capacitor en función del tiempo.
\begin{align*}
Q(t)=Q\sin\left(\omega_0 t + \pi/2\right)
\end{align*}
\item Determinar la corriente en el inductor en función del tiempo.
\begin{align*}
I(t)=I_{\text{max}}\cos\left(\omega_0 t + \pi/2\right)
\end{align*}
\item ¿Cuál es el valor de $\omega_0$?
\item ¿Cuál es la frecuencia de oscilación de la carga?
\end{enumerate}

{\color{red}CIRCUITOS RLC libre (Amortiguado)}

%ejercicio
\item Considere un capacitor $C$ previamente cargado con una carga $\pm Q$ que se conecta en serie a un inductor $L$ y a una resistencia $R$.
\begin{enumerate}
\item Determinar la corriente en el inductor en función del tiempo. Asuma que se tiene amortiguación débil.
\begin{align*}
I(t)&= I_{\text{max}}\, \exp^{-\gamma t} \sin\left(\omega t + \pi/2\right)\\
I(t)&= I_{\text{max}}\, \exp^{-\dfrac{R}{2L}t}\sin\left(\sqrt{\dfrac{1}{LC}-\dfrac{R^2}{4L^2}}t+\pi/2\right)
\end{align*}
\item ¿Cuál es la frecuencia de oscilación de la carga?
\item Grafique la corriente en función del tiempo.
\end{enumerate}

{\color{red}CIRCUITOS RLC  (Forzado)}

%ejercicio
\item Considere un capacitor $C$ que se conecta en serie a un inductor $L$, a una resistencia $R$ y una \textit{fem} variable $\xi=V_0\sin(\omega_f t)$:
\begin{enumerate}
\item Determinar la corriente en el inductor en función del tiempo. 
\begin{align*}
I(t)&= I_{0}\, \sin\left(\omega_f t - \alpha \right)
=\dfrac{V_0}{\sqrt{R^2+\left(\omega_f L - \dfrac{1}{\omega_f C}\right)^2}}\, \sin\left(\omega_f t - \alpha \right)
\end{align*}
\item ¿Cuál es la frecuencia de oscilación de la carga?
\item Grafique la corriente en función del tiempo.
\end{enumerate}
